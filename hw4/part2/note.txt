#find min between two clusters
def min_cluster_distance(c_cluster,x,j):
	distance = 100
	for len(c_cluster[x]):
		for len(c_cluster[j]):
			updated_distance = np.sqrt(sum(np.square(c_cluster[x] - c_cluster[j])))
			if (distance > updated_distance):
				distance = updated_distance
	return distance
